# Vue的初始化过程

在之前的“找到vue的最终引入”篇我们知道Vue的最终引入在`instance/index`

```javascript
import { initMixin } from './init'
import { stateMixin } from './state'
import { renderMixin } from './render'
import { eventsMixin } from './events'
import { lifecycleMixin } from './lifecycle'
import { warn } from '../util/index'

// 核心构造函数
function Vue (options) {
  ...
  this._init(options)
}
// 给vue原型添加方法
initMixin(Vue) // _init
stateMixin(Vue) // $set $delete $watch
eventsMixin(Vue) // $on $once $off $emit
lifecycleMixin(Vue) // _update $forceUpdate $destroy
renderMixin(Vue) // $nextTick _render

export default Vue
```

那么核心应该就在`this._init(options)`里了。在`initMixin(Vue)`里初始化了`_init`方法，在这里我们只留下核心代码：

```javascript
  Vue.prototype._init = function (options?: Object) {
    const vm: Component = this
    // 每个组件的唯一标识
    vm._uid = uid++

    // a flag to avoid this being observed
    vm._isVue = true
    ...
    // 合并option
      vm.$options = mergeOptions(
      	// 当前实例的constructor的options
        resolveConstructorOptions(vm.constructor),
        options || {},
        vm
      )
    ...
    // expose real self
    vm._self = vm
    initLifecycle(vm) // 组件的父子关系
    initEvents(vm) // 组件的事件
    initRender(vm) // vm._c 创建虚拟节点
    callHook(vm, 'beforeCreate') // 调用生命周期
    initInjections(vm) // resolve injections before data/props
    initState(vm) // 状态初始化 核心响应式原理
    initProvide(vm) // resolve provide after data/props
    callHook(vm, 'created') // 调用生命周期
    ...
    if (vm.$options.el) {
      // 挂载 创建一个渲染Watcher 进行渲染组件操作
      vm.$mount(vm.$options.el)
    }
  }
```

`vm.$mount`方法在`runtime/index`文件里

```javascript
import { mountComponent } from 'core/instance/lifecycle'

Vue.prototype.$mount = function (
  el?: string | Element,
  hydrating?: boolean
): Component {
  el = el && inBrowser ? query(el) : undefined
  return mountComponent(this, el, hydrating)
}
```

主要调用了`mountComponent`方法

```javascript
export function mountComponent (
  vm: Component,
  el: ?Element,
  hydrating?: boolean
): Component {
  vm.$el = el
  // 调用生命周期
  callHook(vm, 'beforeMount')

  let updateComponent = () => {
    // 渲染成虚拟dom再更新
    vm._update(vm._render(), hydrating)
  }
  // we set this to vm._watcher inside the watcher's constructor
  // since the watcher's initial patch may call $forceUpdate (e.g. inside child
  // component's mounted hook), which relies on vm._watcher being already defined
  // 创建了一个渲染watcher 执行updateComponent 执行时会去实例取值
  new Watcher(vm, updateComponent, noop, {
    before () {
      if (vm._isMounted && !vm._isDestroyed) {
        callHook(vm, 'beforeUpdate')
      }
    }
  }, true /* isRenderWatcher */)

  // manually mounted instance, call mounted on self
  // mounted is called for render-created child components in its inserted hook
  if (vm.$vnode == null) {
    vm._isMounted = true
    callHook(vm, 'mounted')
  }
  return vm
}

```
